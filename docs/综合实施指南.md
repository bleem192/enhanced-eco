## 💳 功能1：支付功能实现

### 技术方案：模拟支付流程

**为什么选择模拟支付：**
- 课程设计项目不适合接入真实支付网关
- 微信/支付宝需要企业资质和个人认证
- 模拟支付可以完整展示支付流程

### 实现步骤

#### 步骤1：创建支付路由

**文件：** `backend/routes/payment.js`

```javascript
const express = require('express');
const { pool } = require('../config/database');
const { authenticateToken, requireRole } = require('../middleware/auth');
const { successResponse, errorResponse } = require('../middleware/response');
const { sendOrderConfirmation } = require('../utils/email');

const router = express.Router();

// 模拟支付
router.post('/pay/:orderId', authenticateToken, requireRole('customer'), async (req, res) => {
  try {
    const { orderId } = req.params;
    const { payment_method = 'wechat' } = req.body;
    
    // 1. 验证订单存在且属于当前用户
    const [orders] = await pool.query(
      'SELECT * FROM orders WHERE id = ? AND user_id = ? AND status = "pending"',
      [orderId, req.user.id]
    );
    
    if (orders.length === 0) {
      return errorResponse(res, '订单不存在或已支付', 400);
    }
    
    const order = orders[0];
    
    // 2. 模拟支付处理（延迟500ms模拟支付过程）
    await new Promise(resolve => setTimeout(resolve, 500));
    
    // 3. 更新订单状态
    await pool.query(
      'UPDATE orders SET status = "payed", payment_method = ? WHERE id = ?',
      [payment_method, orderId]
    );
    
    // 4. 发送邮件确认
    try {
      await sendOrderConfirmation(req.user.email, order);
    } catch (emailError) {
      console.error('邮件发送失败:', emailError);
      // 邮件失败不影响支付成功
    }
    
    // 5. 记录操作日志
    await pool.query(
      'INSERT INTO operation_logs (user_id, username, operation, module, ip_address) VALUES (?, ?, ?, ?, ?)',
      [req.user.id, req.user.username, `订单支付成功: ${orderId}`, 'payment', req.ip]
    );
    
    return successResponse(res, {
      order_id: orderId,
      status: 'payed',
      payment_method
    }, '支付成功');
    
  } catch (error) {
    console.error('支付失败:', error);
    return errorResponse(res, '支付失败', 500);
  }
});

// 支付状态查询
router.get('/status/:orderId', authenticateToken, async (req, res) => {
  try {
    const { orderId } = req.params;
    
    const [orders] = await pool.query(
      'SELECT id, status, payment_method FROM orders WHERE id = ? AND user_id = ?',
      [orderId, req.user.id]
    );
    
    if (orders.length === 0) {
      return errorResponse(res, '订单不存在', 404);
    }
    
    return successResponse(res, orders[0], '查询成功');
  } catch (error) {
    return errorResponse(res, '查询失败', 500);
  }
});

module.exports = router;
```

```javascript
class EmailService {

    this.transporter = nodemailer.createTransport({
```

```javascript
const express = require('express');
const { pool } = require('../config/database');
const { authenticateToken, requireRole } = require('../middleware/auth');
const { successResponse, errorResponse, paginatedResponse } = require('../middleware/response');

const router = express.Router();

// 获取所有商品类别
router.get('/', async (req, res) => {
  try {
    const [categories] = await pool.query(`
      SELECT 
        category,
        COUNT(*) as product_count,
        COALESCE(SUM(stock), 0) as total_stock,
        COALESCE(AVG(price), 0) as avg_price
      FROM products 
      WHERE status = 'available' 
      GROUP BY category 
      ORDER BY product_count DESC
    `);
    
    return successResponse(res, categories, '查询成功');
  } catch (error) {
    return errorResponse(res, '查询失败', 500);
  }
});

// 添加新类别
router.post('/', authenticateToken, requireRole(['sales', 'admin']), async (req, res) => {
  try {
    const { category_name } = req.body;
    
    if (!category_name || category_name.trim().length === 0) {
      return errorResponse(res, '类别名称不能为空', 400);
    }
    
    const trimmedName = category_name.trim();
    
    // 检查类别是否已存在
    const [existing] = await pool.query(
      'SELECT id FROM products WHERE category = ? LIMIT 1',
      [trimmedName]
    );
    
    if (existing.length > 0) {
      return errorResponse(res, '该类别已存在', 400);
    }
    
    // 记录操作日志
    await pool.query(
      'INSERT INTO operation_logs (user_id, username, operation, module, ip_address) VALUES (?, ?, ?, ?, ?)',
      [req.user.id, req.user.username, `添加商品类别: ${trimmedName}`, 'categories', req.ip]
    );
    
    return successResponse(res, { category: trimmedName }, '类别添加成功');
  } catch (error) {
    return errorResponse(res, '服务器内部错误', 500);
  }
});

// 删除类别
router.delete('/:category', authenticateToken, requireRole(['sales', 'admin']), async (req, res) => {
  try {
    const { category } = req.params;
    
    // 检查该类别下是否有商品
    const [products] = await pool.query(
      'SELECT COUNT(*) as count FROM products WHERE category = ?',
      [category]
    );
    
    if (products[0].count > 0) {
      return errorResponse(res, `该类别下有 ${products[0].count} 个商品，请先删除或移动商品后再试`, 400);
    }
    
    // 记录操作日志
    await pool.query(
      'INSERT INTO operation_logs (user_id, username, operation, module, ip_address) VALUES (?, ?, ?, ?, ?)',
      [req.user.id, req.user.username, `删除商品类别: ${category}`, 'categories', req.ip]
    );
    
    return successResponse(res, null, '类别删除成功');
  } catch (error) {
    return errorResponse(res, '服务器内部错误', 500);
  }
});

module.exports = router;
```

## 🛒 功能4：「浏览该商品的人还购买」推荐功能

### 算法原理

基于购买行为的协同过滤：
1. 找出购买过商品A的所有用户
2. 统计这些用户还购买过哪些其他商品
3. 按共现次数排序，返回Top N

### 实现步骤

#### 步骤1：添加推荐接口

**文件：** `backend/routes/analysis-enhanced.js`

在文件末尾添加：

```javascript
// 「浏览该商品的人还购买」推荐
router.get('/co-purchase/:productId', authenticateToken, async (req, res) => {
  try {
    const { productId } = req.params;
    const { limit = 6 } = req.query;
    
    // 查询购买过该商品的用户
    const [purchasedUsers] = await pool.query(`
      SELECT DISTINCT o.user_id
      FROM orders o
      JOIN order_items oi ON o.id = oi.order_id
      WHERE oi.product_id = ? 
        AND o.status IN ('payed', 'shipped', 'completed')
    `, [productId]);
    
    // 如果没有购买记录，返回热门商品
    if (purchasedUsers.length === 0) {
      const [popular] = await pool.query(`
        SELECT p.id, p.name, p.category, p.price, p.image_url,
          COALESCE(SUM(oi.quantity), 0) as sales_count
        FROM order_items oi
        JOIN products p ON oi.product_id = p.id
        JOIN orders o ON oi.order_id = o.id
        WHERE p.id != ? AND p.status = 'available'
          AND o.status IN ('payed', 'shipped', 'completed')
        GROUP BY p.id
        ORDER BY sales_count DESC
        LIMIT ?
      `, [productId, parseInt(limit)]);
      
      return successResponse(res, {
        products: popular,
        type: 'popular_fallback'
      }, '查询成功');
    }
    
    const userIds = purchasedUsers.map(u => u.user_id);
    
    // 找出这些用户还购买的其他商品
    const [coProducts] = await pool.query(`
      SELECT 
        oi.product_id as id,
        p.name,
        p.category,
        p.price,
        p.image_url,
        COUNT(DISTINCT o.user_id) as co_occurrence_count,
        SUM(oi.quantity) as sales_count
      FROM orders o
      JOIN order_items oi ON o.id = oi.order_id
      JOIN products p ON oi.product_id = p.id
      WHERE o.user_id IN (?)
        AND oi.product_id != ?
        AND o.status IN ('payed', 'shipped', 'completed')
        AND p.status = 'available'
      GROUP BY oi.product_id
      ORDER BY co_occurrence_count DESC, sales_count DESC
      LIMIT ?
    `, [userIds, productId, parseInt(limit)]);
    
    return successResponse(res, {
      products: coProducts,
      type: 'co_purchase'
    }, '查询成功');
    
  } catch (error) {
    console.error('Get co-purchase recommendations error:', error);
    return errorResponse(res, '服务器内部错误', 500);
  }
});
```
