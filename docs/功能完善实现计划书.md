### 3.1 订单分隔问题解决方案

#### 问题描述
不同登录用户应只能查看自己的订单，确保订单数据隔离。

#### 现状分析
查看后端代码 `backend/routes/orders.js`：
```javascript
router.get('/', authenticateToken, async (req, res) => {
  // 第21-24行已正确实现用户订单隔离
  if (req.user.role === 'customer') {
    query += ' WHERE user_id = ?';
    params.push(user_id);
  }
});
```

**结论：** 后端API已正确实现订单隔离，通过 `authenticateToken` 中间件获取用户ID并过滤。

#### 前端验证
查看前端代码 `frontend/src/views/Orders.vue`：
```javascript
const fetchOrders = async () => {
  const response = await api.orders.getOrders()
  if (response.success) {
    orders.value = response.data.list
  }
}
```

**结论：** 前端调用已正确，后端API返回的即为当前用户的订单。

#### 解决方案
1. ✅ **后端已正确实现** - 无需修改
2. ⚠️ **前端需增强** - 添加用户信息显示，验证订单确实属于当前用户
3. 🔧 **前端优化** - 添加订单筛选和搜索功能

### 3.2 支付功能实现

#### 功能描述
用户下单后应完成支付流程，订单状态从 `pending` 变为 `payed`。

#### 实现方案

**方案一：模拟支付（推荐用于课程设计）**

```javascript
// backend/routes/payment.js
router.post('/pay', authenticateToken, requireRole('customer'), async (req, res) => {
  try {
    const { order_id, payment_method } = req.body;
    
    // 验证订单存在且属于当前用户
    const [orderRows] = await pool.query(
      'SELECT * FROM orders WHERE id = ? AND user_id = ? AND status = "pending"',
      [order_id, req.user.id]
    );
    
    if (orderRows.length === 0) {
      return errorResponse(res, '订单不存在或已支付', 400);
    }
    
    // 模拟支付成功
    await pool.query(
      'UPDATE orders SET status = "payed", payment_method = ? WHERE id = ?',
      [payment_method, order_id]
    );
    
    // 发送邮件确认
    await sendOrderConfirmationEmail(req.user.email, orderRows[0]);
    
    return successResponse(res, { order_id }, '支付成功');
  } catch (error) {
    return errorResponse(res, '支付失败', 500);
  }
});
```

### 3.3 邮件确认功能实现

#### 技术选型
使用 Node.js 的 `nodemailer` 库发送邮件

#### 安装依赖
```bash
npm install nodemailer
```

#### 实现代码

```javascript
// backend/utils/email.js
const nodemailer = require('nodemailer');

const transporter = nodemailer.createTransport({
  service: 'gmail', // 或使用其他邮件服务
  auth: {
    user: process.env.EMAIL_USER,
    pass: process.env.EMAIL_PASS
  }
});

const sendOrderConfirmation = async (to, order) => {
  const mailOptions = {
    from: process.env.EMAIL_USER,
    to,
    subject: '订单确认 - 您的订单已成功支付',
    html: `
      <h2>订单确认</h2>
      <p>订单号：${order.id}</p>
      <p>订单金额：¥${order.total_amount}</p>
      <p>支付方式：${order.payment_method}</p>
      <p>收货地址：${order.shipping_address}</p>
      <p>感谢您的购买！</p>
    `
  };
  
  return transporter.sendMail(mailOptions);
};

module.exports = { sendOrderConfirmation };
```

### 3.4 分类管理API实现

#### 功能需求
销售人员能够添加新类别、删除已有类别。

#### API设计

```javascript
// backend/routes/categories.js

// 1. 添加商品类别
router.post('/', authenticateToken, requireRole(['sales', 'admin']), async (req, res) => {
  try {
    const { category_name } = req.body;
    
    if (!category_name || category_name.trim() === '') {
      return errorResponse(res, '类别名称不能为空', 400);
    }
    
    // 检查类别是否已存在
    const [existing] = await pool.query(
      'SELECT * FROM products WHERE category = ? LIMIT 1',
      [category_name]
    );
    
    if (existing.length > 0) {
      return errorResponse(res, '该类别已存在', 400);
    }
    
    // 记录操作日志
    await pool.query(
      'INSERT INTO operation_logs (user_id, username, operation, module, ip_address) VALUES (?, ?, ?, ?, ?)',
      [req.user.id, req.user.username, `添加商品类别: ${category_name}`, 'categories', req.ip]
    );
    
    return successResponse(res, { category: category_name }, '类别添加成功');
  } catch (error) {
    return errorResponse(res, '服务器内部错误', 500);
  }
});

// 2. 删除商品类别
router.delete('/:category', authenticateToken, requireRole(['sales', 'admin']), async (req, res) => {
  try {
    const { category } = req.params;
    
    // 检查该类别下是否有商品
    const [products] = await pool.query(
      'SELECT COUNT(*) as count FROM products WHERE category = ?',
      [category]
    );
    
    if (products[0].count > 0) {
      return errorResponse(res, `该类别下有 ${products[0].count} 个商品，无法删除`, 400);
    }
    
    // 记录操作日志
    await pool.query(
      'INSERT INTO operation_logs (user_id, username, operation, module, ip_address) VALUES (?, ?, ?, ?, ?)',
      [req.user.id, req.user.username, `删除商品类别: ${category}`, 'categories', req.ip]
    );
    
    return successResponse(res, null, '类别删除成功');
  } catch (error) {
    return errorResponse(res, '服务器内部错误', 500);
  }
});

// 3. 获取所有类别
router.get('/', async (req, res) => {
  try {
    const [categories] = await pool.query(
      'SELECT DISTINCT category, COUNT(*) as product_count FROM products GROUP BY category'
    );
    return successResponse(res, categories, '查询成功');
  } catch (error) {
    return errorResponse(res, '服务器内部错误', 500);
  }
});
```

### 3.5 「浏览该商品的人还购买」推荐功能实现

#### 算法原理
基于用户行为数据的协同过滤：当用户浏览某商品时，找出同样浏览过该商品的其他用户，然后推荐这些用户购买过的其他商品。

#### 实现步骤

**步骤1：创建用户浏览-购买关联表**

```sql
-- 在 database-enhanced.js 中添加
CREATE TABLE IF NOT EXISTS product_co_purchase (
  id INT PRIMARY KEY AUTO_INCREMENT,
  source_product_id INT NOT NULL,
  target_product_id INT NOT NULL,
  co_occurrence_count INT DEFAULT 1,
  score DECIMAL(5, 4) DEFAULT 0,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (source_product_id) REFERENCES products(id),
  FOREIGN KEY (target_product_id) REFERENCES products(id),
  UNIQUE KEY unique_co_occurrence (source_product_id, target_product_id)
);
```

**步骤2：实现推荐算法**

```javascript
// backend/routes/analysis-enhanced.js

// 「浏览该商品的人还购买」推荐接口
router.get('/co-purchase/:productId', authenticateToken, async (req, res) => {
  try {
    const { productId } = req.params;
    const { limit = 6 } = req.query;
    
    // 查询购买了当前商品的用户
    const [purchasedUsers] = await pool.query(`
      SELECT DISTINCT o.user_id
      FROM orders o
      JOIN order_items oi ON o.id = oi.order_id
      WHERE oi.product_id = ? AND o.status IN ('payed', 'shipped', 'completed')
    `, [productId]);
    
    if (purchasedUsers.length === 0) {
      // 如果没有购买记录，返回热门商品
      const [popular] = await pool.query(`
        SELECT p.id, p.name, p.category, p.price, p.image_url,
          SUM(oi.quantity) as sales_count
        FROM order_items oi
        JOIN products p ON oi.product_id = p.id
        WHERE p.id != ? AND p.status = 'available'
        GROUP BY p.id
        ORDER BY sales_count DESC
        LIMIT ?
      `, [productId, parseInt(limit)]);
      
      return successResponse(res, { products: popular }, '查询成功');
    }
    
    const userIds = purchasedUsers.map(u => u.user_id);
    
    // 找出这些用户还购买的其他商品
    const [coProducts] = await pool.query(`
      SELECT 
        oi.product_id,
        p.name,
        p.category,
        p.price,
        p.image_url,
        COUNT(DISTINCT o.user_id) as co_occurrence_count,
        SUM(oi.quantity) as sales_count
      FROM orders o
      JOIN order_items oi ON o.id = oi.order_id
      JOIN products p ON oi.product_id = p.id
      WHERE o.user_id IN (?) 
        AND oi.product_id != ?
        AND o.status IN ('payed', 'shipped', 'completed')
        AND p.status = 'available'
      GROUP BY oi.product_id
      ORDER BY co_occurrence_count DESC, sales_count DESC
      LIMIT ?
    `, [userIds, productId, parseInt(limit)]);
    
    return successResponse(res, { products: coProducts }, '查询成功');
  } catch (error) {
    console.error('Get co-purchase recommendations error:', error);
    return errorResponse(res, '服务器内部错误', 500);
  }
});
```
